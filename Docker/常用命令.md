### 数据库初始化与连接

```Makefile
.PHONY: mysql_init mysql_cli redis_init redis_cli data_init

APP_NAME = biz-review

DATABASE_NAME = biz_review

MYSQL_ROOT_PASSWORD = root123

data_init : mysql_init  redis_init

# mysql 环境
mysql_init:
	docker run -d \
      --name ${APP_NAME}-mysql \
      -e MYSQL_ROOT_PASSWORD=${MYSQL_ROOT_PASSWORD}\
      -e MYSQL_DATABASE=${DATABASE_NAME} \
      -p 3306:3306 \
      -v ${APP_NAME}-mysql-data:/var/lib/mysql \
      mysql:8.4;
	@echo 'mysql 容器启动成功'
# 进入 mysql 客户端
mysql_cli:
	docker run -it --network host --rm mysql:8.4 mysql --default-character-set=utf8mb4 -h127.0.0.1 -P3306 -uroot -p

# redis 环境
redis_init:
	docker run -d \
      --name ${APP_NAME}-redis \
      -p 6379:6379 \
      -v ${APP_NAME}-redis-data:/data \
      redis:7.2 \
      redis-server --appendonly yes
	@echo 'redis 容器启动成功'
# 进入 redis 客户端
redis_cli:
	docker run -it --network host --rm redis:7.2 redis-cli -h 127.0.0.1 -p 6379
```
